113  // Remove text between brackets using regex

291  std::string transcribe(const std::vector<float>& pcmf32, float& prob,
292  int64_t& t_ms) {
293  const auto t_start = std::chrono::high_resolution_clock::now();
294 
295  prob = 0.0f;
296  t_ms = 0;
297 
298  std::vector<whisper_token> prompt_tokens;
299 
300  whisper_full_params wparams =
301  whisper_full_default_params(WHISPER_SAMPLING_GREEDY);
302 
303  prompt_tokens.resize(1024);
304  prompt_tokens.resize(whisper_tokenize(
305  ctx_wsp, prompt.c_str(), prompt_tokens.data(), prompt_tokens.size()));
306 
307  wparams.print_progress = false;
308  wparams.print_special = params.print_special;
309  wparams.print_realtime = false;
310  wparams.print_timestamps = !params.no_timestamps;
311  wparams.translate = params.translate;
312  wparams.no_context = true;
313  wparams.single_segment = true;
314  wparams.max_tokens = params.max_tokens;
315  wparams.language = params.language.c_str();
316  wparams.n_threads = params.n_threads;
317 
318  wparams.prompt_tokens =
319  prompt_tokens.empty() ? nullptr : prompt_tokens.data();
320  wparams.prompt_n_tokens = prompt_tokens.empty() ? 0 : prompt_tokens.size();
321 
322  wparams.audio_ctx = params.audio_ctx;
323  wparams.speed_up = params.speed_up;
324 
325  if (whisper_full(ctx_wsp, wparams, pcmf32.data(), pcmf32.size()) != 0) {
326  return "";
327  }
328 
329  int prob_n = 0;
330  std::string result;
331 
332  const int n_segments = whisper_full_n_segments(ctx_wsp);
333  for (int i = 0; i < n_segments; ++i) {
334  const char* text = whisper_full_get_segment_text(ctx_wsp, i);
335 
336  result += text;
337 
338  const int n_tokens = whisper_full_n_tokens(ctx_wsp, i);
339  for (int j = 0; j < n_tokens; ++j) {
340  const auto token = whisper_full_get_token_data(ctx_wsp, i, j);
341 
342  prob += token.p;
343  ++prob_n;
344  }
345  }
346 
347  if (prob_n > 0) {
348  prob /= prob_n;
349  }
350 
351  const auto t_end = std::chrono::high_resolution_clock::now();
352  t_ms =
353  std::chrono::duration_cast<std::chrono::milliseconds>(t_end - t_start)
354  .count();
355 
356  return result;
357  }
