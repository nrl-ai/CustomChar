183 std::string LLM::GetAnswer(std::vector<llama_token>& embd) {
184  bool done = false;
185  int last_length = 0;
186  int loop_count = 0;
187  std::string text_to_speak;
188  while (true) {
189  if (embd.size() > 0) {
190  if (n_past_ + (int)embd.size() > n_ctx_) {
191  n_past_ = n_keep_;
192  // Insert n_left/2 tokens at the start of embd from last_n_tokens
193  embd.insert(embd.begin(),
194  embd_inp_.begin() + embd_inp_.size() - n_prev_,
195  embd_inp_.end());
196  // Stop saving session if we run out of context
197  path_session_ = "";
198  }
199 
200  // Try to reuse a matching prefix from the loaded session instead of
201  // re-eval (via n_past) REVIEW
202  if (n_session_consumed_ < (int)session_tokens_.size()) {
203  size_t i = 0;
204  for (; i < embd.size(); i++) {
205  if (embd[i] != session_tokens_[n_session_consumed_]) {
206  session_tokens_.resize(n_session_consumed_);
207  break;
208  }
209 
210  n_past_++;
211  n_session_consumed_++;
212 
213  if (n_session_consumed_ >= (int)session_tokens_.size()) {
214  i++;
215  break;
216  }
217  }
218  if (i > 0) {
219  embd.erase(embd.begin(), embd.begin() + i);
220  }
221  }
222 
223  if (embd.size() > 0 && !path_session_.empty()) {
224  session_tokens_.insert(session_tokens_.end(), embd.begin(), embd.end());
225  n_session_consumed_ = session_tokens_.size();
226  }
227 
228  if (llama_eval(ctx_llama_, embd.data(), embd.size(), n_past_,
229  n_threads_)) {
230  fprintf(stderr, "%s : failed to eval\n", __func__);
231  exit(1);
232  }
233  }
234 
235  embd_inp_.insert(embd_inp_.end(), embd.begin(), embd.end());
236  n_past_ += embd.size();
237 
238  embd.clear();
239 
240  if (done) break;
241 
242  {
243  // out of user input, sample next token
244  const float top_k = 10000;
245  const float top_p = 0.80f;
246  const float temp = 0.20f;
247  const float repeat_penalty = 1.1f;
248  const int repeat_last_n = 256;
249 
250  if (!path_session_.empty() && need_to_save_session_) {
251  need_to_save_session_ = false;
252  llama_save_session_file(ctx_llama_, path_session_.c_str(),
253  session_tokens_.data(), session_tokens_.size());
254  }
255 
256  llama_token id = 0;
257 
258  {
259  auto logits = llama_get_logits(ctx_llama_);
260  auto n_vocab = llama_n_vocab(ctx_llama_);
261 
262  logits[llama_token_eos()] = 0;
263 
264  std::vector<llama_token_data> candidates;
265  candidates.reserve(n_vocab);
266  for (llama_token token_id = 0; token_id < n_vocab; token_id++) {
267  candidates.emplace_back(
268  llama_token_data{token_id, logits[token_id], 0.0f});
269  }
270 
271  llama_token_data_array candidates_p = {candidates.data(),
272  candidates.size(), false};
273 
274  // apply repeat penalty
275  const float nl_logit = logits[llama_token_nl()];
276 
277  llama_sample_repetition_penalty(
278  ctx_llama_, &candidates_p,
279  embd_inp_.data() + std::max(0, n_past_ - repeat_last_n),
280  repeat_last_n, repeat_penalty);
281 
282  logits[llama_token_nl()] = nl_logit;
283 
284  if (temp <= 0) {
285  // Greedy sampling
286  id = llama_sample_token_greedy(ctx_llama_, &candidates_p);
287  } else {
288  // Temperature sampling
289  llama_sample_top_k(ctx_llama_, &candidates_p, top_k, 1);
290  llama_sample_top_p(ctx_llama_, &candidates_p, top_p, 1);
291  llama_sample_temperature(ctx_llama_, &candidates_p, temp);
292  id = llama_sample_token(ctx_llama_, &candidates_p);
293  }
294  }
295 
296  if (id != llama_token_eos()) {
297  // add it to the context
298  embd.push_back(id);
299  text_to_speak += llama_token_to_str(ctx_llama_, id);
300  printf("%s", llama_token_to_str(ctx_llama_, id));
301  }
302  }
303 
304  {
305  std::string last_output;
306  for (int i = embd_inp_.size() - 16; i < (int)embd_inp_.size(); i++) {
307  last_output += llama_token_to_str(ctx_llama_, embd_inp_[i]);
308  }
309  last_output += llama_token_to_str(ctx_llama_, embd[0]);
310 
311  for (std::string& antiprompt : antiprompts_) {
312  if (last_output.find(antiprompt.c_str(),
313  last_output.length() - antiprompt.length(),
314  antiprompt.length()) != std::string::npos) {
315  done = true;
316  text_to_speak = common::Replace(text_to_speak, antiprompt, "");
317  fflush(stdout);
318  need_to_save_session_ = true;
319  break;
320  }
321  }
322  }
323 
324  // Break to avoid infinite loop
325  // TODO: Fix this bug
326  if ((int)text_to_speak.length() == last_length + 1 &&
327  text_to_speak[text_to_speak.length() - 1] == '\n') {
328  ++loop_count;
329  } else {
330  loop_count = 0;
331  }
332  if (loop_count > 5) {
333  break;
334  }
335  last_length = text_to_speak.length();
336  }
337 
338  return text_to_speak;
