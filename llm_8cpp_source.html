177  n_past = n_keep;
178  // Insert n_left/2 tokens at the start of embd from last_n_tokens
179  embd.insert(embd.begin(), embd_inp.begin() + embd_inp.size() - n_prev,
180  embd_inp.end());

182  path_session = "";
183  }
184 
185  // Try to reuse a matching prefix from the loaded session instead of
186  // re-eval (via n_past) REVIEW
187  if (n_session_consumed < (int)session_tokens.size()) {
188  size_t i = 0;
189  for (; i < embd.size(); i++) {
190  if (embd[i] != session_tokens[n_session_consumed]) {
191  session_tokens.resize(n_session_consumed);
192  break;
193  }
194 
195  n_past++;
196  n_session_consumed++;
197 
198  if (n_session_consumed >= (int)session_tokens.size()) {
199  i++;
200  break;
201  }
202  }
203  if (i > 0) {
204  embd.erase(embd.begin(), embd.begin() + i);
205  }
206  }
207 
208  if (embd.size() > 0 && !path_session.empty()) {
209  session_tokens.insert(session_tokens.end(), embd.begin(), embd.end());
210  n_session_consumed = session_tokens.size();
211  }
212

214  fprintf(stderr, "%s : failed to eval\n", __func__);
215  exit(1);
216  }
217  }
218

220  n_past += embd.size();

246

248

250  candidates.reserve(n_vocab);
251  for (llama_token token_id = 0; token_id < n_vocab; token_id++) {
252  candidates.emplace_back(
253  llama_token_data{token_id, logits[token_id], 0.0f});

261

266

268

279  }
